DayOfYear& DayOfYear::operator++()
{
	if (this->day == 365)
		this->day = 1;

		this->day++;
	return *this;
}

DayOfYear DayOfYear::operator++(int)
{
	DayOfYear tmp(*this);
	operator++();
	return tmp;
}

DayOfYear& DayOfYear::operator--()
{
	if (this->day == 1)
		this->day = 365;
	else
		this->day--;
	return *this;
}

DayOfYear DayOfYear::operator--(int)
{
	DayOfYear tmp(*this);
	operator--();
	return tmp;
