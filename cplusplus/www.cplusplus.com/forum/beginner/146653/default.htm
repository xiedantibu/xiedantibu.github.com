#include <type_traits>
#include <windows.h>

template<typename T>
typename std::enable_if<std::is_integral<T>::value && sizeof(T) == sizeof(DWORD),
LONG>::type foo(const T*)
{}

int main()
{
    int* x;
    foo(x); //Works
    unsigned long* l;
    foo(l); //Works
    float f;
    foo (f); //Does not work, f is not integral
    long long* d;
    foo(d); //Does not work, size mismatch
    short* s;
    foo(s); //Does not work, size mismatch
}
