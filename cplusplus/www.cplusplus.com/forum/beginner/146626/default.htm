#include <iostream>
#include "util.h" // display()
#include "str.h" // strFill(), strLength

using namespace std;

void letterToAlpha(char* letter, char* alphabet, char* voidalphabet){
    while(*alphabet){
        if(*letter == *alphabet){
            *voidalphabet = *alphabet;
        }
        alphabet++;
        voidalphabet++;
    }
}

int main(int argc, char *argv[]) {

    char x[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    char y[strLength(x)];
    char c = 'C';

    y[strLength(x) - 1] = '\0';

    strFill(y, '_');

    letterToAlpha(&c, x, y);

    display("\nFull alpha: ");
    display(x);
    display("\n\nVoid alpha: ");
    display(y);
    display("\n\nFull alpha length: ");
    display(strLength(x));
    display("\n\nVoid alpha length: ");
    display(strLength(y));

    return 0;
}
