#include <iostream>
#include <stack>

template<typename T>
class double_stack_queue
{
    std::stack<T> S1;
    std::stack<T> S2;
public:
	void enqueue(const T& in)
	{
	    S1.push(in);
	}

	T dequeue() //Does not works like standard queue
	{
		while(!S1.empty()) {
			S2.push(S1.top());
			S1.pop();
		}
		T temp = S2.top();
		S2.pop();
		while(!S2.empty()) {
			S1.push(S2.top());
			S2.pop();
		}
		return temp;
	}
};

int main()
{
	double_stack_queue<int> foo;
	foo.enqueue(5);
	foo.enqueue(7);
	foo.enqueue(10);

	std::cout << foo.dequeue();
}
