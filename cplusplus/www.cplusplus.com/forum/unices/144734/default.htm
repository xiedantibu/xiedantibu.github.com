double extract(const string & buf, string from, string to)
{
    double result = 0;
    
    size_t pos1 = buf.find(from);
    if (pos1 != string::npos)
    {
        pos1 += from.size();
        size_t pos2 = buf.find(to, pos1);
        if (pos2 != string::npos)
        {
            istringstream ss (buf.substr(pos1, pos2-pos1));
            ss >> result;
        }
    }    
    
    return result;
}
