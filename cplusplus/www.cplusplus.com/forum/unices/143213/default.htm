int FilterEvent(wxEvent& event) {
    if (event.GetEventType() == wxEVT_KEY_DOWN) {
        wxMessageBox(wxString::Format(wxT("%d"), ((wxKeyEvent &)event).GetKeyCode()), wxT("KeyCode"));
        return true;
    }
    return -1;
}
