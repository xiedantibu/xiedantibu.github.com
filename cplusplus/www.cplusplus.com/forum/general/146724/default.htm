template <typename T>
int BST<T>::nodesCount()
{
   int count = 0;
   nodesCount(mRootNode, count);
   return count;
}


//private function
template <typename T>
void nodesCount(BST_Node<T> *node, int &count)
{
   if(node == NULL)
      return;
   else
      count = 1 + nodesCount(node->mLeft, count) + nodesCount(node->mRight, count);
}
