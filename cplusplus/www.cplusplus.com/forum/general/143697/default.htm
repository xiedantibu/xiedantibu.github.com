#include <iostream>
#include <string>
#include <cctype>

using namespace std;

int main() {

    string name;

    while (true) {

        bool ok = true;

        cout << "Enter your name:\n";

        getline(cin, name);

        size_t nameSize = name.size();

        if (nameSize == 0) continue;

        for (size_t i = 0; i < nameSize; ++ i) {
            if (
                !(
                    isalpha(name[i]) ||
                    name[i] == ' ' // || ...
                 )
               ) {
                ok = false; break;
            }
        }

        if (ok) break;
    }

    cout << "Hello, " << name << "!" << endl;

    return 0;
}
