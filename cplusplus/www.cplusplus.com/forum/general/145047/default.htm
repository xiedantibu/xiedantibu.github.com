using namespace std;
int main()
{
    const double PI = 3.14159;
    double radius=0.0;
    double area=0.0;
    double length=0.0;
    double width=0.0;
    double base=0.0;
    double height=0.0;
    int choice=0;
	

    cout << "Geometry Calculator\n";
    cout << "\n";
    cout << "1. Calculate the area of a Circle\n";
    cout << "2. Calculate the area of a Rectangle\n";
    cout << "3. Calculate the area of a Triangle\n";
    cout << "4. Quit\n\n";
    cout << "Enter your choice (1-4): "; 
	cin >> choice; 

    
    if (choice>=1 && choice<=4)

        switch (choice)
        {
            case 1: //Area of Circle
                cout << "\n";
                cout << "Enter the circle's radius: ";
                cin >> radius;
                
                if (radius>0)
                {    
                    area = PI * radius * radius;
                    //cout << fixed << setprecision(4);
                    cout << "The area is " << area;
                }
                else
                {
                    cout << "The radius can not be less than or equal zero.";
                }
                break;
            case 2: //Area of Rectangle
                cout << "\n";
                cout << "Enter the rectangle's length: ";
                cin >> length;
                cout << "Enter the rectangle's width: ";
                cin >> width;
                
                if (length>0 && width>0 && length>width)
                {    
                    area = length * width;
                    cout << "The area is " << area;
                }
                else
                {
                    cout << "\nOnly enter positive values for length and width and length must be greater than width.";
                }
            
                break;
            case 3: //Area of Triangle

                cout << "Enter the triangle's height: ";
